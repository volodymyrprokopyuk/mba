\startenvironment layout

%% language
\mainlanguage[es]

%% colors
\setupcolor[hex]
\definecolor[Base][h=184be5]
\definecolor[Light][h=4e74e5]
\definecolor[Lighter][h=91a8eb]
\definecolor[Dark][h=0732b4]
\definecolor[Darker][h=052278]
\definecolor[Gray][h=888888]

%% symbols
\definefontsynonym[Awesome][name:fontawesome]
\definesymbol[AwItem][\getnamedglyphdirect{Awesome}{ok}]
\define\AwItem{{\Gray\symbol[AwItem]}}
\definesymbol[AwRArrow][\getnamedglyphdirect{Awesome}{arrow_right}]
\define\AwRArrow{{\Gray\symbol[AwRArrow]}}
\definesymbol[AwRArrowLong][\getnamedglyphdirect{Awesome}{long_arrow_right}]
\define\AwRArrowLong{{\Gray\symbol[AwRArrowLong]}}
\definesymbol[AwCaretRight][\getnamedglyphdirect{Awesome}{caret_right}]
\define\AwCaretRight{{\Gray\symbol[AwCaretRight]}}

%% SourcePro font
\starttypescript[sans][SourcePro]
\definefontsynonym[Sans][name:sourcesansprolight]
\definefontsynonym[SansBold][name:sourcesansprobold]
\definefontsynonym[SansItalic][name:sourcesansprolightit]
\definefontsynonym[SansBoldItalic][name:sourcesansproit]
\stoptypescript
\definetypeface[SourcePro][ss][sans][SourcePro]
%% math font
\definetypeface[SourcePro][mm][math][iwona]
\setupbodyfont[SourcePro,sans,14pt]

%% layout
\setuplayout[backspace=80pt,edge=0pt,edgedistance=5pt,
  margin=70pt,margindistance=5pt,width=fit,
  topspace=40pt,top=35pt,topdistance=5pt,header=20pt,headerdistance=10pt,
  footer=20pt,footerdistance=10pt,bottom=40pt,bottomdistance=5pt,height=fit]
\setupwhitespace[big]

%% header and footer
\startsectionblockenvironment[frontpart]
\setuppagenumbering[location=]
\stopsectionblockenvironment
\setuppagenumbering[location={footer,right}]
\setupheadertexts[{\hfill\Gray\getmarking[chapter]}]

%% headings
\setuphead[chapter][number=no,style=\bfc\Gray]
\setuphead[section][style=\bfb\Gray]
\setuphead[subsection][style=\bfa\Gray]
\setupcaptions[minwidth=0.8\textwidth,location={bottom},align=middle,
  headstyle=\tf,style=\bf,color=Gray]

%% lists
\definesymbol[1][{\AwItem}]
\definesymbol[2][{\AwCaretRight}]

%% interaction
\setupinteraction[state=start,style=normal,color=Base]

%% table of contents
\setupcombinedlist[content][alternative=c]
\setuplist[chapter][style=bold]

%% images
\setupexternalfigures[directory={../tex/img}]

%% flow charts
\usemodule[chart]
\setupFLOWcharts[maxwidth=\textwidth,dx=1\bodyfontsize,
  maxheight=\textheight,dy=1\bodyfontsize,offset=0pt]
\setupFLOWshapes[rulethickness=2pt,
  framecolor=Base,background=color,backgroundcolor=Lighter]
\setupFLOWlines[rulethickness=3pt,color=Base]

%% Comment
\define[1]\Comment{\startalignment[flushright]\Gray #1\stopalignment\page[no]}

%% Paragraph
\define[1]\Paragraph{{\bf\Gray #1}\page[no]}

%% Source
\define[1]\Source{\rightaligned{\Gray Fuente: {[}\in{}[#1]{]}}\page[no]}

\stopenvironment
